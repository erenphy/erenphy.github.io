phase_5汇编代码分析

08048e22 <phase_5>:
 8048e22:    53                       push   %ebx
 8048e23:    83 ec 18                 sub    $0x18,%esp
 8048e26:    8b 5c 24 20              mov    0x20(%esp),%ebx
 8048e2a:    89 1c 24                 mov    %ebx,(%esp)
 8048e2d:    e8 69 02 00 00           call   804909b <string_length>

 #len(string) == 6,否则bomb
 8048e32:    83 f8 06                 cmp    $0x6,%eax
 8048e35:    74 05                    je     8048e3c <phase_5+0x1a>
 8048e37:    e8 89 03 00 00           call   80491c5 <explode_bomb>

 #edx = 0; eax = 0;||int x=0;int y=0
 8048e3c:    ba 00 00 00 00           mov    $0x0,%edx
 8048e41:    b8 00 00 00 00           mov    $0x0,%eax
    循环体：
     #ecx = eax+ebx = y+input;||temp = y+ebx
     8048e46:    0f b6 0c 03              movzbl (%ebx,%eax,1),%ecx

     #ecx 取低4位相当于取模
     8048e4a:    83 e1 0f                 and    $0xf,%ecx
     8048e4d:    03 14 8d 80 a2 04 08     add    0x804a280(,%ecx,4),%edx
  • edx = 0x804a280初始地址 + ecx的offset 的累加
    (gdb) print *0x804a280@20
    $4 = {2, 10, 6, 1, 12, 16, 9, 3, 4, 7, 14, 5, 11, 8, 15, 13, 2032168787, 1948284271, 1802398056, 1970239776}
#eax++,进入循环体，直至==6||while y<6;
8048e54:    83 c0 01                 add    $0x1,%eax
8048e57:    83 f8 06                 cmp    $0x6,%eax
8048e5a:    75 ea                    jne    8048e46 <phase_5+0x24>

#edx == 31,否则bomb
8048e5c:    83 fa 1f                 cmp    $0x1f,%edx
8048e5f:    74 05                    je     8048e66 <phase_5+0x44>
8048e61:    e8 5f 03 00 00           call   80491c5 <explode_bomb>
8048e66:    83 c4 18                 add    $0x18,%esp
8048e69:    5b                       pop    %ebx
8048e6a:    c3                       ret

  • 要使得x=0的初始值累加6次之后的值==31,在以上内存中符合条件的一组是66661;其下标为3和4找到mod16 == 3 和 4的ascii可表示字符作为输入

  • 验证

补充：

  • movzbl指令负责拷贝一个字节，并用0填充其目的操作数中的其余各位，这种扩展方式叫“零扩展”。
  • movsbl指令负责拷贝一个字节，并用源操作数的最高位填充其目的操作数中的其余各位，这种扩展方式叫“符号扩展”。
